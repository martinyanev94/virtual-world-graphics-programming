#version 450

layout(location = 0) in vec3 inPosition;
layout(location = 1) in mat4 inInstanceMatrix;

layout(location = 0) out vec3 fragColor;

layout(set = 0, binding = 0) uniform BufferObject {
    mat4 modelMatrix;
};

void main() {
    gl_Position = modelMatrix * inInstanceMatrix * vec4(inPosition, 1.0);
    fragColor = vec3(inPosition.x * 0.5 + 0.5, inPosition.y * 0.5 + 0.5, 0.0);
}
vkCmdDrawIndexed(commandBuffer, indexCount, treeInstances.instanceCount, 0, 0, 0);
